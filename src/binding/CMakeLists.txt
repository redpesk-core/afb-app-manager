if(libsystemd_FOUND AND AFB_FOUND AND NOT ONLY_SDK)
	MESSAGE(STATUS "Creating afm-binding")

	pkg_get_variable(VSCRIPT afb-binding version_script)

	add_library(
		afm-binding
		MODULE
		afm-binding.c
		afm-udb.c
		afm-urun.c
	)
	target_compile_options(
		afm-binding
		PRIVATE
		${AFB_CFLAGS}
	)
	target_include_directories(
		afm-binding
		PRIVATE
		${AFB_INCLUDE_DIRS}
	)
	target_link_libraries(
		afm-binding
		secwrp
		utils
		${AFB_LIBRARIES}
	)
	target_link_directories(
		afm-binding
		PRIVATE
		${AFB_LIBRARY_DIRS}
	)
	set_target_properties(
		afm-binding
		PROPERTIES
		PREFIX ""
		LINK_FLAGS "-Wl,--version-script=${VSCRIPT}"
	)
	install(
		TARGETS afm-binding
		LIBRARY
		DESTINATION ${afm_libexecdir}
	)
else()
	MESSAGE(STATUS "Not creating afm-binding")
endif()
